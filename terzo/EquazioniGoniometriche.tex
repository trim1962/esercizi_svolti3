	\begin{align*}

x=&\frac{\ang{170}}{2}+k\frac{\ang{360}}{2}\\
x=&\ang{85}+k\ang{180}\\
	\end{align*}
	
	La soluzione è
	\[x=\ang{85}+k\ang{180}\]

\begin{align*}

\end{exercise}

&\begin{cases}
x+y-3=0\\

\end{cases}
&&\begin{cases}
x-3x+1-3=0\\
y=-3x+1
\end{cases}\\
&\begin{cases}
-2x-2=0\\
y=-3x+1
\end{cases}
&&\begin{cases}
y=-1\\
y=-3(-1)+1
\end{cases}\\
&\begin{cases}
y=4\\
x=-1
\end{cases}
\end{align*}
